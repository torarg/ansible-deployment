 18class DeploymentVault(AnsibleDeployment):
 19    """
 20    Represents a deployment vault used for file encryption.
 21
 22    Args:
 23        vault_files (sequence): Sequence of paths defining vault content.
 24        path (path): Vault root directory.
 25
 26    Attributes:
 27        path (Path): Vault root directory.
 28        locked (bool): True if vault is currently locked.
 29        new_key (bool): True if a new key was created.
 30        key_file (Path): Path to key file.
 31                         Defaults to `self.path / 'deployment.key'`
 32        key (byte): Key used for encryption.
 33        files (sequence): Sequence of paths defining vault content.
 34
 35    Note:
 36        If `key_file` is not present at initialization, it will be created.
 37    """
 38
 39    encryption_suffix = ".enc"
 40    key_file_name = "deployment.key"
 41    tar_file_name = "deployment.tar.gz"
 42    lock_file_name = ".LOCKED"
 43
 44    def __init__(self, vault_files, path, key=None):
 45        self.new_key = False
 46        self.locked = False
 47        self.path = Path(path)
 48        self.git_path = self.path / '.git'
 49        self.shadow_git_path = self.path / '.git.shadow'
 50        self.lock_file_path = self.path / self.lock_file_name
 51        self.tar_path = self.path / self.tar_file_name
 52        self.encrypted_tar_path = Path(str(self.tar_path) + self.encryption_suffix)
 53        self.encrypted_tar_sha256sum_path = Path(str(self.encrypted_tar_path) + ".SHA256")
 54        self.encrypted_tar_sha256sum = None
 55        self.key_file = self.path / self.key_file_name
 56        self._load_key(key)
 57        self.files = vault_files
 58        self.locked_files = list(
 59            self.path.glob("**/*{}".format(self.encryption_suffix))
 60        )
 61        if self.lock_file_path.exists():
 62            self.locked = True
 63            self.files = self.locked_files
 64
 65    def _load_key(self, key=None):
 66        """
 67        Read or create `self.key_file` and update `self.key`.
 68
 69        Note:
 70            Key will be generated and written to `self.key_file` if
 71            `self.key_file` does not exist.
 72        """
 73        if key is not None:
 74            self.key = key
 75            self._save_key()
 76        elif self.key_file.exists():
 77            with open(self.key_file, "rb") as fobj:
 78                self.key = fobj.read()
 79        else:
 80            self.key = self._generate_key()
 81            self._save_key()
 82            self.new_key = True
 83
 84    def _save_key(self):
 85        """
 86        Write `self.key` to `self.key_file`.
 87        """
 88        with open(self.key_file, "wb") as fobj:
 89            fobj.write(self.key)
 90        self.key_file.chmod(0o600)
 91
 92    @staticmethod
 93    def _generate_key():
 94        """
 95        Generate a new key.
 96
 97        Returns:
 98            byte: Generated key.
 99        """
100        return Fernet.generate_key()
101
102    def _encrypt_file(self, file_path):
103        """
104        Encrypt a single file.
105
106        Args:
107            file_path (Path): Path object for target file.
108        """
109        cipher_suite = Fernet(self.key)
110        with open(file_path, "rb") as fobj:
111            plain_data = fobj.read()
112        if int(oct(file_path.stat().st_mode)[-3]) < 6:
113            file_path.chmod(0o640)
114        with open(file_path, "wb") as fobj:
115            fobj.write(cipher_suite.encrypt(plain_data))
116        file_path.replace(str(file_path) + self.encryption_suffix)
117
118    def _sha256sum(self, file_path):
119        """
120        Create sha256 sum of file and write it to ``file_path.SHA256``.
121        """
122        h  = hashlib.sha256()
123        b  = bytearray(128*1024)
124        mv = memoryview(b)
125        with open(file_path, 'rb', buffering=0) as f:
126            for n in iter(lambda : f.readinto(mv), 0):
127                h.update(mv[:n])
128        return h.hexdigest()
129
130    def _encrypt_files(self, files):
131        """
132        Encrypt a sequence of files.
133
134        Args:
135            files (sequence): Sequence of Path objects.
136        """
137        with tarfile.open(self.tar_path, 'w:gz') as tar:
138            for file_name in files:
139                file_path = Path(file_name)
140                if file_path.exists() and file_path.name != self.key_file.name:
141                    tar.add(file_path)
142        self._encrypt_file(self.tar_path)
143        self.encrypted_tar_sha256sum = self._sha256sum(self.encrypted_tar_path)
144
145    def _decrypt_files(self, files):
146        """
147        Decrypt a sequence of files.
148
149        Args:
150            files (sequence): Sequence of Path objects.
151        """
152        for file_name in files:
153            file_path = Path(file_name)
154            if not file_path.exists():
155                pass
156            elif file_path.is_file():
157                self._decrypt_file(file_path)
158            elif file_path.is_dir():
159                files_in_subdir = list(file_path.glob("*"))
160                self._decrypt_files(files_in_subdir)
161
162    def _decrypt_file(self, file_path):
163        """
164        Decrypt a single file.
165
166        Args:
167            file_path (Path): Path object for target file.
168        """
169        cipher_suite = Fernet(self.key)
170        unencrypted_file_path = str(file_path)[: -len(self.encryption_suffix)]
171        with open(file_path, "rb") as fobj:
172            encrypted_data = fobj.read()
173        with open(unencrypted_file_path, "wb") as fobj:
174            fobj.write(cipher_suite.decrypt(encrypted_data))

176    def setup_shadow_repo(self, remote_config):
177        """
178        Replaces deployment repo with a 'shadow repo'.
179
180        The deployment git repository in ``self.path / '.git'``
181        will be moved to ``self.path / '.git.enc'`` and if
182        ``self.path / '.git.shadow'`` exists, it's config will be moved
183        to ``self.path / '.git'``.
184        A new repositoriy will be intialized and all files in ``self.files``
185        including the original deployment repository will be
186        added and commited to this new repository.
187
188        The purpose of the shadow repository is to have
189        a pushable deployment state while the deployment
190        is locked.
191
192        Args:
193            remote_config (RepoConfig): RepoConfig Namedtuple.
194        """
195        exclude_files = ('deployment.key', '.terraform', 'deployment.tar.gz.enc', '.ssh')
196        include_files = ('.LOCKED', '.drone.yml', '.gitlab-ci', '.gitignore')
197
198
199        shutil.rmtree(self.git_path)
200        if self.shadow_git_path.exists():
201            shutil.move(self.shadow_git_path, self.git_path)
202
203        deployment_files = list(
204            self.path.glob("[!.]*")
205        )
206        for file_name in include_files:
207            deployment_files.append(self.path / file_name)
208
209        shadow_repo_files = []
210        for file in deployment_files:
211            if file.name not in exclude_files and file.exists():
212                shadow_repo_files.append(file)
213        shadow_repo = DeploymentRepo(self.path, files=shadow_repo_files, remote_config=remote_config)
214        shadow_repo.blobs['deployment_data'] = self.encrypted_tar_path
215        shadow_repo.init()
216
217        shadow_repo.update(
218            message="Shadow repository activated.",
219            force_commit=False,
220            files=shadow_repo_files
221        )
222
223    def _restore_deployment_dir(self):
224        """
225        Restores deployment dir from tar archive.
226        """
227
228        if self.git_path.exists():
229            shutil.move(self.git_path, self.shadow_git_path)
230
231        with tarfile.open(self.tar_path) as tar:
232            tar.extractall(self.path)
233
234        self.tar_path.unlink()

237    def lock(self):
238        """
239        Encrypts all files stored in the vault and activates shadow repo.
240
241        The git repository in ``self.path`` will be encrypted and replaced
242        with a shadow repository containing all encrypted vault files as
243        a single commit. If an old instance of the shadow repository
244        had a custuom config this will be restored in the new shadow
245        repository.
246        """
247        if not self.locked:
248            self._encrypt_files(self.files)
249            self._write_hash()
250            self.locked = True
251            self.lock_file_path.touch()
252        else:
253            raise DeploymentVaultError("Deployment already locked")
254
255    def _write_hash(self):
256        """
257        Write sha256sum to file.
258        """
259        with open(self.encrypted_tar_sha256sum_path, 'w') as f:
260            f.write(self.encrypted_tar_sha256sum)
261
262    def _verify_hash(self, file_path, expected_sha256sum):
263        """Verify sha256 sum of file against an expected value."""
264        sha256sum = self._sha256sum(file_path)
265        return expected_sha256sum == sha256sum
266
267    def unlock(self, force_unlock=False):
268        """
269        Decrypts all vault files and restores the deployment repository.
270
271        The shadow repository's configuration file willl be saved
272        to ``self.path / '.git.shadow/config``.
273        """
274        if self.locked:
275            with open(self.encrypted_tar_sha256sum_path) as f:
276                expected_hash = f.read()
277            valid_hash = self._verify_hash(self.encrypted_tar_path, expected_hash)
278            if not valid_hash and not force_unlock:
279                raise DeploymentVaultError("Verification of encrypted deployment failed")
280            self._decrypt_file(self.encrypted_tar_path)
281            self._restore_deployment_dir()
282            self.locked = False
283            self.lock_file_path.unlink()
284        else:
285            raise DeploymentVaultError("Deployment already unlocked")
286
287    def delete(self, delete_shadowgit=False):
288        self.lock_file_path.unlink(missing_ok=True)
289        self.encrypted_tar_path.unlink(missing_ok=True)
290        self.encrypted_tar_sha256sum_path.unlink(missing_ok=True)
291        self.tar_path.unlink(missing_ok=True)
292        if delete_shadowgit:
293            shutil.rmtree(self.shadow_git_path, ignore_errors=True)

Replaces deployment repo with a 'shadow repo'.

The deployment git repository in self.path / '.git' will be moved to self.path / '.git.enc' and if self.path / '.git.shadow' exists, it's config will be moved to self.path / '.git'. A new repositoriy will be intialized and all files in self.files including the original deployment repository will be added and commited to this new repository.

The purpose of the shadow repository is to have a pushable deployment state while the deployment is locked.
